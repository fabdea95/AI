(defclass persona
        (is-a USER)
        (role concrete)
        (pattern-match reactive)
        (slot nombre (type STRING))
        (slot edad (type INTEGER))
        (slot peso (type INTEGER))
        (slot altura (type INTEGER))
	(slot presion_sanguinea (type FLOAT))
)

(defclass problema_salud
        (is-a PRIMITIVE)
	(role concrete)
        (pattern-match reactive)
        (slot dolor_espalda (allowed-values 0 1) (default 0))
        (slot dolor_articular (allowed-values 0 1) (default 0))
        (slot cervical (allowed-values 0 1) (default 0))
        (slot escoliosis (allowed-values 0 1) (default 0))
        (slot lordosis (allowed-values 0 1) (default 0))
        (slot cifosis (allowed-values 0 1) (default 0))
        (slot dolor_lombar (allowed-values 0 1) (default 0))
        (slot tendinitis (allowed-values 0 1) (default 0))
)

(defclass dieta
	(is-a PRIMITIVE)
	(role concrete)
	(pattern-match reactive)
	(slot consumo_fruta (allowed-values muy_bajo bajo normal elevado) (default normal))
	(slot consumo_verdura (allowed-values muy_bajo bajo normal elevado) (default normal))
	(slot consumo_hidcarb (allowed-values muy_bajo bajo normal elevado) (default normal))
	(slot consumo_proteinas (allowed-values muy_bajo bajo normal elevado) (default normal))
	(slot consumo_grasa (allowed-values muy_bajo bajo normal elevado) (default normal))
	(slot abuso_sal (allowed-values 0 1) (default 0))
	(slot picar (allowed-values nunca cada_vez a_menudo) (default cada_vez))
)

(defclass habitos
	(is-a PRIMITIVE)
	(role abstract)
	(slot frecuencia (allowed-values muy_baya baya normal elevada))
	(slot duracion (type INTEGER))
)

(defclass trabajo
	(is-a habitos)
	(role concrete)
	(pattern-match reactive)
	(slot sentado (allowed-values 0 1))
	(slot de_pie (allowed-values 0 1))
	(slot desplaz_pie (allowed-values 0 1))
	(slot mov_repet (allowed-values 0 1))
	(slot levant_peso (allowed values 0 1))
	(slot esfuerzo_musc (allowed values 0 1))
)
(defclass tareas_domest
	(is-a habitos)
	(role concrete)
	(pattern-match reactive)
	(slot planchar (allowed-values 0 1))
	(slot barrer (allowed-values 0 1))
	(slot fregar (allowed-values 0 1))
	(slot colada (allowed-values 0 1))
)
(defclass desplaziamentos
	(is-a habitos)
	(role concrete)
	(pattern-match reactive)
